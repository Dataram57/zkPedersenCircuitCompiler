

template Multiply(a, b){
    // a * b = ((a + b)**2 - a**2 - b**2) / 2

    //(a + b)**2
    ab_sum += [
        a,
        b
    ];
    square ab_sum_sq(ab_sum);

    //-(a**2)
    square a_sq(a);
    negation a_sq_neg(a_sq);

    //-(b**2)
    square b_sq(b);
    negation b_sq_neg(b_sq);

    //(a + b)**2 - a**2 - b**2)
    sum += [
        ab_sum_sq,
        a_sq_neg,
        b_sq_neg
    ];

    //((a + b)**2 - a**2 - b**2) / 2
    out += [
        [1/2, sum]
    ];
};
